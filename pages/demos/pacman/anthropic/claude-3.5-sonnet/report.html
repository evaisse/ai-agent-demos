<!DOCTYPE html>
<html>
<head>
    <title>PAC-MAN</title>
    <style>
        body {
            background-color: black;
            display: flex;
            justify-content: center;
            align-items: center;
            height: 100vh;
            margin: 0;
        }
        
        #gameBoard {
            border: 2px solid blue;
        }

        .score {
            color: white;
            font-family: Arial, sans-serif;
            font-size: 20px;
            position: absolute;
            top: 10px;
            left: 10px;
        }
    </style>
</head>
<body>
    <div class="score">Score: <span id="score">0</span></div>
    <canvas id="gameBoard"></canvas>

    <script>
        const canvas = document.getElementById('gameBoard');
        const ctx = canvas.getContext('2d');
        const scoreElement = document.getElementById('score');

        canvas.width = 560;
        canvas.height = 600;

        class Pacman {
            constructor(x, y) {
                this.x = x;
                this.y = y;
                this.radius = 15;
                this.speed = 3;
                this.direction = 0; // 0: right, 1: up, 2: left, 3: down
                this.mouthOpen = 0;
                this.mouthSpeed = 0.15;
            }

            draw() {
                ctx.save();
                ctx.translate(this.x, this.y);
                ctx.rotate((this.direction * 90) * Math.PI / 180);
                
                ctx.beginPath();
                ctx.arc(0, 0, this.radius, this.mouthOpen * Math.PI, (2 - this.mouthOpen) * Math.PI);
                ctx.lineTo(0, 0);
                ctx.fillStyle = 'yellow';
                ctx.fill();
                
                ctx.restore();
            }

            update() {
                this.mouthOpen += this.mouthSpeed;
                if (this.mouthOpen > 0.5 || this.mouthOpen < 0) {
                    this.mouthSpeed = -this.mouthSpeed;
                }
            }

            move() {
                switch(this.direction) {
                    case 0: this.x += this.speed; break;
                    case 1: this.y -= this.speed; break;
                    case 2: this.x -= this.speed; break;
                    case 3: this.y += this.speed; break;
                }

                // Keep Pacman within bounds
                if (this.x < this.radius) this.x = this.radius;
                if (this.x > canvas.width - this.radius) this.x = canvas.width - this.radius;
                if (this.y < this.radius) this.y = this.radius;
                if (this.y > canvas.height - this.radius) this.y = canvas.height - this.radius;
            }
        }

        class Dot {
            constructor(x, y) {
                this.x = x;
                this.y = y;
                this.radius = 3;
                this.eaten = false;
            }

            draw() {
                if (!this.eaten) {
                    ctx.beginPath();
                    ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
                    ctx.fillStyle = 'white';
                    ctx.fill();
                }
            }
        }

        const pacman = new Pacman(100, 100);
        let dots = [];
        let score = 0;

        // Create dots grid
        for (let i = 40; i < canvas.width; i += 40) {
            for (let j = 40; j < canvas.height; j += 40) {
                dots.push(new Dot(i, j));
            }
        }

        function checkCollision() {
            dots.forEach(dot => {
                if (!dot.eaten) {
                    const dx = pacman.x - dot.x;
                    const dy = pacman.y - dot.y;
                    const distance = Math.sqrt(dx * dx + dy * dy);
                    
                    if (distance < pacman.radius + dot.radius) {
                        dot.eaten = true;
                        score += 10;
                        scoreElement.textContent = score;
                    }
                }
            });
        }

        document.addEventListener('keydown', (event) => {
            switch(event.key) {
                case 'ArrowRight':
                    pacman.direction = 0;
                    break;
                case 'ArrowUp':
                    pacman.direction = 1;
                    break;
                case 'ArrowLeft':
                    pacman.direction = 2;
                    break;
                case 'ArrowDown':
                    pacman.direction = 3;
                    break;
            }
        });

        function gameLoop() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            // Draw dots
            dots.forEach(dot => dot.draw());
            
            // Update and draw Pacman
            pacman.update();
            pacman.move();
            pacman.draw();
            
            // Check for collisions
            checkCollision();
            
            requestAnimationFrame(gameLoop);
        }

        gameLoop();
    </script>
