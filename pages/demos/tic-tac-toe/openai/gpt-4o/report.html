        let currentPlayer = 'X';

        function handleClick(e) {
            const index = e.target.dataset.index;
            if (board[index] !== null) return;

            board[index] = currentPlayer;
            e.target.textContent = currentPlayer;

            if (checkWin()) {

            const winPatterns = [

                [2, 5, 8],
