        let currentPlayer = 'X';

        let gameOver = false;

        function handleCellClick(cell) {
            const index = cell.getAttribute('data-index');
            if (board[index] === '' && !gameOver) {
                board[index] = currentPlayer;
                cell.textContent = currentPlayer;
