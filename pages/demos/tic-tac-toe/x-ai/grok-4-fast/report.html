    <script>
        const board = Array(9).fill(null);
        const cells = document.querySelectorAll('.cell');
        const status = document.getElementById('status');
        const resetBtn = document.getElementById('reset');
        let currentPlayer = 'X';
        let gameOver = false;

        const winningCombos = [
            [0, 1, 2], [3, 4, 5], [6, 7, 8], // rows
            [0, 3, 6], [1, 4, 7], [2, 5, 8], // columns
            [0, 4, 8], [2, 4, 6] // diagonals
        ];

        function handleClick(event) {
            const cell = event.target;
            const index = parseInt(cell.dataset.index);

            if (board[index] || gameOver) return;

            board[index] = currentPlayer;
            cell.textContent = currentPlayer;
            cell.classList.add(currentPlayer.toLowerCase());

            const winCombo = checkWin();
            if (winCombo) {
                status.textContent = Player ${currentPlayer} wins!;
                highlightWinningCells(winCombo);
                gameOver = true;
                document.getElementById('board').classList.add('game-over');
                return;
            }

            if (board.every(cell => cell !== null)) {
                status.textContent = "It's a draw!";
                applyDrawEffect();
                gameOver = true;
                document.getElementById('board').classList.add('game-over');
                return;
            }

            currentPlayer = currentPlayer === 'X' ? 'O' : 'X';
            status.textContent = Player ${currentPlayer}'s turn;
        }

        function checkWin() {
            for (let combo of winningCombos) {
                const [a, b, c] = combo;
                if (board[a] && board[a] === board[b] && board[a] === board[c]) {
                    return combo;
                }
            }
            return null;
        }

        function highlightWinningCells(combo) {
            combo.forEach(index => {
                cells[index].classList.add('winning');
            });
        }

        function applyDrawEffect() {
            cells.forEach(cell => cell.classList.add('draw'));
        }

        function resetGame() {
            board.fill(null);
            cells.forEach(cell => {
                cell.textContent = '';
                cell.classList.remove('x', 'o', 'winning', 'draw');
            });
            currentPlayer = 'X';
            gameOver = false;
            status.textContent = "Player X's turn";
            document.getElementById('board').classList.remove('game-over');
        }

        cells.forEach(cell => cell.addEventListener('click', handleClick));
        resetBtn.addEventListener('click', resetGame);
    </script>
